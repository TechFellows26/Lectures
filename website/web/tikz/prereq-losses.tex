  tick style={lightfg},
  ticklabel style={color=lightfg},
  label style={color=lightfg},

  legend style={fill=darkbg, draw=bordercol, text=lightfg, font=\small},
  grid style={draw=bordercol, opacity=0.4},
  xlabel={prediction $\hat{y}$},
  ylabel={loss},
  title={Loss Functions ($y = 1$)},
  domain=-3:3,
  samples=120,
  xmin=-3, xmax=3,

  every axis plot/.style={very thick},

% Squared loss: (1 - yhat)^2
\addplot[accent, solid]
  {(1 - x)^2};
\addlegendentry{Squared: $(1 - \hat{y})^2$}

% Logistic loss: ln(1 + exp(-yhat))
\addplot[bluecol, dashed]
  {ln(1 + exp(-x))};
\addlegendentry{Logistic: $\ln(1 + e^{-\hat{y}})$}

% 0-1 loss: step function at yhat = 1
\addplot[redcol, dotted, domain=-3:1, samples=2] {1};
\addplot[redcol, dotted, domain=1:3, samples=2] {0};
\addplot[redcol, only marks, mark=*, mark size=2.5pt] coordinates {(1,1)};
\addplot[redcol, only marks, mark=o, mark size=2.5pt, thick] coordinates {(1,0)};
\addlegendentry{0-1 loss}
